 67     /**
 68      * Ask the user a yes and no question, in a confirm dialog box
 69      * 
 70      * @param text the yes no question to ask in the dialog
 71      * @return true (yes), false (no)
 72      */
 73     public static boolean yesNo(String text) {
 74         // Create graphical display area with formatted text to put in dialog
 75         JTextArea area = formatArea(text);
 76         // Store the user's response in a variable from what they clicked on
 77         // when the dialog appears only showing "yes" and "no" buttons for 
 78         // the user to choose from  
 79         int response = JOptionPane.showConfirmDialog(null,area,title,
 80                 JOptionPane.YES_NO_OPTION,style);
 81         if (response == JOptionPane.YES_OPTION) return true;
 82         else                                    return false;
 83     }

 85     /**
 86      * Asks the user for a string input in a input dialog box
 87      * 
 88      * @param text the text for the dialog box
 89      * @return a valid string
 90      */
 91     public static String input(String text) {
 92         // Create graphical display area with formatted text to put in dialog
 93         JTextArea area = formatArea(text);
 94         // Store the user's response in a variable from what they typed into
 95         // a input dialog
 96         String value = JOptionPane.showInputDialog(null,area,title,style);
 97         // Force a loop if the user left the dialog empty and clicked "ok" or
 98         // they clicked "cancel" or the "X"
 99         while (value == null || value.equals("")) {
100             // Recreate the graphical display area
101             area  = formatArea(ERROR_1 + text);
102             value = JOptionPane.showInputDialog(null,area,title,style);
103         }
104         return value;           // Once they have entered a value, return it
105     }

140     /**
141      * Asks the user for a number (integer) in a input dialog box
142      * 
143      * @param text the text for the dialog box
144      * @param minimum the lowest value in the input range
145      * @param maximum the highest value in the input range
146      * @return a valid integer
147      */
148     public static int inputInteger(String text, int minimum, int maximum) {
149         String value = input(text);                 // Get user's response
150         int number = 0;                             // Variable to store value
151         boolean done = false;                       // Flag to keep trying
152         while (!done) {                             // Loop until valid
153             if (Numbers.isInteger(value) == false) {    // Not a valid integer
154                 value = input(ERROR_2 + text);      // Input again with message
155             }
156             else {
157                 number = Integer.parseInt(value);   // Convert to integer
158                 if (Numbers.inRange(number, minimum, maximum)) {
159                     done = true;                    // Number is in range
160                 }
161                 else {
162                     value = input(ERROR_3 + text); // Not in range, input again
163                 }
164             }            
165         }
166         return number;                          // Return valid integer
167     }

257      * @param text the string of text to format
258      * @return the formatted text area for display
259      */
260     public static JTextArea formatArea(String text) {
261         // Create a graphics object to display the graphical things
262         JTextArea area = new JTextArea();
263         // Assign the global variable font, colors, and passed text to the area
264         area.setFont(font);
265         area.setBackground(background);
266         area.setForeground(foreground);
267         area.setText(text);
