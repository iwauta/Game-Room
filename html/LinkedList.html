122      * Inserts (adds) data (generic type) into the back (tail, end, etc) of
123      * this list

126      * @return the operation was successful (true) or not (false)

128     public boolean addBack(T data) {
129         if (data == null) return false;     // null data cannot be added            
130         Node<T> node = new Node<>(data);    // new (memory) Node object created
131         
132         // General programmer scenarios to consider:
133         //  (1) Typical case - what does a 'normal' user do?
134         //  (2) Edge case(s) - what if we pushing the edge/limits?
135         //  (3) Beyond the edge - what if we try to intentially crash this?
136         
137         // Scenarios for this method to consider:
138         //  (1) Empty list
139         //  (2) List of 1 or more nodes
140         
141         if (isEmpty()) {                    // Adding first node
142             head = tail = node;             // Set LinkedList references
143         }
144         else {                              // Subsequent nodes added
145             node.previous = tail;           // Link node to rest of list
146             tail.next = node;               // Connect rest of list to node
147             tail = node;                    // Reassign tail reference
148         }
149         length++;                           // Increase length environmental
150         return true;                        // Operation successful
151     }

320      * Removes (deletes) the first (head) node of the list
321      * 
322      * @return the data in the first node (or null)
323      */
324     public T removeFront() {
325         if (isEmpty()) return null;             // no front to remove
326         T data = front();                       // store head data
327         if (length == 1) finalize();            // 1 node list, wipe list
328         else {                
329             head = head.next;                   // advanced head reference
330             head.previous.next = null;          // cut old head reference
331             head.previous = null;               // cut reference to old head
332             length--;                           // reduce list length
333             System.gc();                        // call system garbage collector
334         }
335         return data;                            // return stored data
336     }

363     public T remove(int index) {
364         if (!inRange(index))   return null;             // not in range
365         if (index == 0)        return removeFront();    // remove first
366         if (index == length-1) return removeBack();     // remove last
367         Node current = getNode(index);                  // get to index
368         current.next.previous = current.previous;       // change references
369         current.previous.next = current.next;
370         current.next = current.previous = null;        
371         length--;                                       // reduce list length
372         return (T)current.data;                         // return index data
373     }

382         if (data == null) return false;         // invalid data to search for
383         Node current = head;                    // start reference at head
384         while (current != null) {               // traverse list

441     public int numberOf(T data) {
442         if (data == null) return 0;             // reject null data
443         int counter = 0;                        // start a counter
444         Node current = head;                    // start at head of list
445         while (current != null) {               // traverse list
446             if (current.data.equals(data)) {    // item found in list

454     /**
455      * Accesses all occurrences of the passed data in the list and returns an
456      * integer array containing all index values the data occurred at
457      * 
458      * @param data the data to search for
459      * @return all indices location in an array or null if no indices
460      */
461     public int[] allIndices(T data) {
462         if (data == null)    return null;       // reject null data
463         if (!contains(data)) return null;       // no data in the list
464         int size = numberOf(data);              // get number of occurrences
465         int[] array = new int[size];            // create array 
466         Node current = head;                    // start at head
467         int counter = 0;                        // start counter
468         for (int i = 0; i < length; i++) {      // traverse list
469             if (current.data.equals(data)) {    // item encountered
470                 array[counter] = i;             // insert index into array
471                 counter++;                      // increase counter
472                 if (counter >= size) {
473                     return array;
474                 }
475             }
476             current = current.next;             // move to next node
477         }
478         return array;                           // return completed array
479     }

693      */
694     public final void fromLinkedList(LinkedList<T> list) {
695         if (list == null) return;                   // error check
696         finalize();                                 // wipe list memory
697         for (int i = 0; i < list.size(); i++) {     // traverse list
698             add(list.get(i));                       // get and add item
699         }
700     }

702     /**
703      * Returns an array that contains the same data as the list
704      * 
705      * @param array the data type array
706      * @return an array of generic type T
707      */
708     public T[] toArray(T[] array) {
709         if (array == null) return null;
710         array = (T[])(
711                 Array.newInstance(
712                         array.getClass().getComponentType(), 
713                         length)
714                 );                              // create empty array
715         for (int i = 0; i < length; i++) {      // traverse list
716             array[i] = get(i);                  // add to array
717         }
718         return array;                           // return completed array
719     }

722      * Accessor method to the encapsulated (private) property of the first
723      * (head) node of the list
724      * 
725      * @return reference to the first node
726      */
727     protected Node getFirstNode() {
728         return head;
729     }

732      * Accessor method to the encapsulated (private) property of the last
733      * (tail) node of the list
734      * 
735      * @return reference to the last node
736      */
737     protected Node getLastNode() {
738         return tail;
739     }

741     /**
742      * Accessor method to retrieve a node reference (link, pointer to) for this
743      * index location
744      * 
745      * @param index the subscript, location, index value
746      * @return a reference to the node at this location or null 
747      */
748     protected Node getNode(int index) {
749         if (!inRange(index))   return null;             // not valid index
750         if (index == 0)        return getFirstNode();   // first node returned     
751         if (index == length-1) return getLastNode();    // last node returned
752         Node current = head;                            // start at first node
753         for (int i = 0; i < index; i++) {               // move to index
754             current = current.next;                     // advance reference
755         }
756         return current;                                 // return reference
757     }

759     /**
760      * Error checks to see if the index is in the range of the list
761      * 
762      * @param index the location (element number) to check
763      * @return it is in-range (true) or not (false)
764      */
765     private boolean inRange(int index) {
766         if (isEmpty())       return false;  // Empty list no valid index
767         if (index < 0)       return false;  // index before first valid number
768         if (index >= length) return false;  // index after last valid number
769         return true;                        // index is valid
770     }
