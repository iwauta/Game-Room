10 /**
11  * NumberGuessing.java - main structure of Numer Guessing. User gets different 
12  * number of guesses corresponds to a level they choose
13  *
14  * @author u.iwai
15  * @since 2023/01/27, 12:52:55
16  */
17 public class NumberGuessing extends MiniGames
18 {
19 
20     // Properties......
21     private static final double EASY_RATE = 1.2;
22     private static final double NORMAL_RATE = 1.5;
23     private static final double HARD_RATE = 1.7;
24     private static int count;
25     private static int number;
26     private static int userInput;
27     // Range of the rangom number
28     private static final int MIN = 1;
29     private static final int MAX = 20;
30     
31     /**
32      * Constructor, set class properties
33      * 
34      * @param num use's choice of level 0(easy) or 1(normal) or 2(hard)
35      */
36     public NumberGuessing(int num) {
37         super(1);
38         // Sets the rate depanding on the level
39         if(num == 0){
40             super.setRate(EASY_RATE);
41             count = 6;
42         }
43         else if(num == 1){
44             super.setRate(NORMAL_RATE);
45             count = 5;
46         }
47         else{
48             super.setRate(HARD_RATE);
49             count = 3;
50         }
51     }
52     
53     /**
54      * Main logic of the game
55      */
56     public static void run() {
57         // Generates a random number between MIN and MAX as an opponent's choice
58         number = Numbers.random(MIN, MAX);
59         String text = "\tEnter your guess\n\tRange: " + MIN 
60                 + " - " + MAX + "\n\n\t" + count + " guess left";
61         String text2= "";
62         userInput = Dialogs.inputInteger(text, MIN, MAX);
63         // user guessed incorrectly
64         while(count >= 1 && userInput != number) {
65             // reduce guesses remaining
66             count --;
67             text = "\tEnter your guess\n\tRange: " + MIN 
68                 + " - " + MAX + "\n\n\t" + count + " guess left";
69             String result = "Your guess: " + userInput;
70             if (userInput > number) text2 = "\n\nYour guess is too big!\n";
71             else                    text2 = "\n\nYour guess is too small!\n";
72             // user guesses again
73             userInput = Dialogs.inputInteger
74                         (result + text2 + text, MIN, MAX);
75         }
76         // Game set (user guessed correctly or used up their guesses)
77         FinalProject.showResult();
78     }
79     
80     /**
81      * Determines if the user wins or loses
82      * 
83      * @return user wins (true) or user loses (false)
84      */
85     @Override
86     public boolean didIWin() {
87         String result = "Your guess: " + userInput;
88         // user guessed correctly
89         if(userInput == number){
90             Dialogs.output(result + "\n\nCongrats! You guessed right!");
91             return true;
92         }
93         // user didn't guess correctly
94         Dialogs.output(result + "\n\nYou have no guesses left!\n\nThe number was: " + number);
95         return false;
96     }

98 }
