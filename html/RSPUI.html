 95     /**
 96      * user moves the mouse to select their choice
 97      * 
 98      * @param evt changes the label so that it tells the user's choice
 99      */
100     private void jLabel1MouseMoved(java.awt.event.MouseEvent evt) {                                   
101         // Sets text for jLabel3 depending on the location of the mouse
102         if (evt.getY() > 0 && evt.getY() < 130) {
103             if (evt.getX() > 0 && evt.getX() <= 86) { // rock
104                 jLabel3.setText("Rock");
105                 userChoice = 1;
106             } else if (evt.getX() > 86 && evt.getX() <= 172) { // scissors
107                 jLabel3.setText("Scissors");
108                 userChoice = 2;
109             } else if (evt.getX() > 172 && evt.getX() < 240) { // paper
110                 jLabel3.setText("Paper");
111                 userChoice = 3;
112             } else {
113                 jLabel3.setText("Click on the picture");
114             }
115         } else {
116             jLabel3.setText("Click on the picture");
117         }
118 
119 
120     }

122     /**
123      * user clicks on a picture, sets a valure for the user's choices
124      * 
125      * @param evt make it invisible, go to the result display
126      */
127     private void jLabel1MouseClicked(java.awt.event.MouseEvent evt) {                                     
128         // Confirms the user's choice
129         String message
130         = "Your choice: " + jLabel3.getText() + "\n\nClick OK to see the result";
131         // Clicked on the picture
132         if (evt.getY() > 40 && evt.getY() < 130) {
133             // rock
134             if (evt.getX() > 0 && evt.getX() <= 86) { 
135                 userChoice = 1;
136                 jLabel3.setText("Rock");
137             } // scissors
138             else if (evt.getX() > 86 && evt.getX() <= 172) { 
139                 userChoice = 2;
140                 jLabel3.setText("Scissors");
141             } // paper
142             else if (evt.getX() > 172 && evt.getX() < 240) { 
143                 userChoice = 3;
144                 jLabel3.setText("Paper");
145             }
146         }
147         JOptionPane.showMessageDialog(null, message);
148         this.setVisible(false);
149         RSP.result();
150     }
