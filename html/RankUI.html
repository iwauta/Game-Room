162     private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
163         // TODO add your handling code here:
164         this.setVisible(false);
165         // User wants to keep playing, go back to select screen
166         if (Dialogs.yesNo("Do you want to keep playing?" + 
167                 "\n\nBy saying 'NO' you will go back to the home screen!")) {
168             // remove the player from the list so that they don't occupy multiple 
169             // spots of the list
170             FinalProject.players.remove(FinalProject.player);
171             FinalProject.play();
172         } // User doesn't want to keep playing, go back to welcome screen
173         else{
174             Dialogs.output("Thank you for using " + Dialogs.title);
175             FinalProject.welcome();
176         }
