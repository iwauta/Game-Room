  6 /**
  7  * Node.java - a representation of a generic node for use with
  8  * a LinkedList object, and could be "visualized" as:
  9  * 
 10  *                      NODE
 11  *                +--------------+
 12  * previous <---- |              |
 13  *                |     data     |
 14  *                |   (generic)  |
 15  *                |              |----> next
 16  *                +--------------+
 17  *
 18  * @author u.iwai
 19  * @param <T> The generic data type used in the class
 20  * @since 13-Dec-2022
 21  */
 22 public class Node <T>
 23 {
 24 
 25     /** The generic data type stored */
 26     public T data;
 27     
 28     /** 
 29      * Self reference (pointer,link) to the next node that 'will' be
 30      * in a 'chain' (linked list)
 31      */
 32     public Node next;
 33     
 34     /** Link / pointer / reference to the previous (back) node */
 35     public Node previous;
 36     
 37     
 38     /**
 39      * Class constructor, sets the class properties
 40      * 
 41      * @param data the node data (generic)
 42      * @param next reference to the next Node object
 43      * @param previous reference to the previous Node object
 44      */
 45     public Node(T data, Node next, Node previous) {
 46         this.data     = data;
 47         this.next     = next;
 48         this.previous = previous;
 49         // We assign the parameters of this method to the properties of this 
 50         // class (the global variables of the class)        
 51     }
 52     
 53     /**
 54      * Class constructor, sets the class properties
 55      * 
 56      * @param data the node data (generic)
 57      * @param next reference to the next Node object
 58      */
 59     public Node(T data, Node next) {
 60         this.data     = data;
 61         this.next     = next;
 62         this.previous = null;       
 63     }

 67      * Class constructor, sets the class properties
 68      * 
 69      * @param data the node data (generic)

 71     public Node(T data) {

 74         this.previous = null;       
 75     }

 87     /**
 88      * String representation of this object
 89      *
 90      * @return The object represented as a String
 91      */
 92     @Override
 93     public String toString() {
 94         String text = "";
 95         
 96         // Display the data stored in the node (or null)
 97         if (data == null) text += "null";               // Invalid data 
 98         else              text += data.toString();      // Generic has toString
 99                 
100         // Display the connection to the next node
101         text += "\t";
102         if      (next      == null) text += "n -> null";
103         else if (next.data == null) text += "n -> node";
104         else                        text += "n -> " + next.data.toString();
105         
106         // Display the connection to the previous node 
107         text += "\t";
108         if      (previous      == null) text += "p -> null";
109         else if (previous.data == null) text += "p -> node";
110         else                            text += "p -> " + previous.data.toString();
111                 
112         // Add the memory address of this node
113         text += "\t" + super.toString();
114         
115         // Return all the data compiled
116         return text;
117     }

120      * Deep comparison, determines if two objects are "equal" in this context
121      *
122      * @param object the object to compare to
123      * @return the objects are "equal" (true) or not (false)

125     @Override

127         // Error check first for nulls appearing which could cause crashes
128         if (this.data == null) return false;    // Data inside this object
129         if (object    == null) return false;    // Parameter sent to this method
130         // Convert (cast) the object into a "Node"
131         Node that = (Node)object;
132         if (that.data == null) return false;    // Data inside the parameter
133         // Now compare both data properties
134         return this.data.equals(that.data);

147      * a Deep clone, creates a duplicate object using new memory
148      *
149      * @return a "clone" of the object using new memory
150      */
151     @Override
152     public Node clone() {
153         return new Node(this.data, this.next, this.previous);
154         // Annoynmous clone object created and returned
155     }

157     /**
158      * Frees up all memory used by this object, and it is called when this 
159      * object is "deleted" or "removed" or "destroyed"
160      */
161     @Override
162     public void finalize() {
163         data = null;
164         next = previous = null;     // Simplified line of code
165         System.gc();                // Runs the garbage collector in Java
166         // The 'garbage collector' is how Java manages memory. References to 
167         // 'memory' locations (e.g. variables, objects, etc.) that do not 
168         // contain actual data (for example a null) can be considered 'garbage'
169         // wastes of memory and should be 'cleaned up'. This means the memory 
170         // can be 'freed up' so other parts of the program could use that 
171         // memory and the memory isn't wasted. Java does do this automatically
172         // (like C# it has "automatic garbage collection") when it 'finds the 
173         // time' in the execution cycle, but you can make Java do it 
174         // explicitly by running the line above.
175     }
