42     /**
43      * Determines if the user wins or loses
44      * 
45      * @return user wins (true) or user loses (false)
46      */
47     @Override
48     public boolean didIWin() {
49         user = RSPUI.getUserChoice(); 
50         if      (user == 1 && opp == 2) return true; // user-rock, opp-scissors
51         else if (user == 2 && opp == 3) return true; // user-scissors, opp-paper
52         else if (user == 3 && opp == 1) return true; // user-paper, opp-rock
53         return false;                                // user loses
54     }

56     /**

58      * the user to play again.
59      */
60     public static void result() {
61         String u = "";
62         String o = "";
63         user = RSPUI.getUserChoice();     
64         if (user == 1) u = "ROCK";
65         if (user == 2) u = "SCISSORS";
66         if (user == 3) u = "PAPER";
67         if (opp == 1)  o = "ROCK";
68         if (opp == 2)  o = "SCISSORS";
69         if (opp == 3)  o = "PAPER";
70         String text = "You picked: " + u + "\nBot picked: " + o;
71         // Outputs result
72         Dialogs.output(text);
73         // Ties up, force the user to play again
74         if (user == opp) {

76             user = 0;
77             opp = getOpponentChoice();
78             RSPUI rspUI = new RSPUI();
