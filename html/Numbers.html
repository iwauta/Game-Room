 20     /**
 21      * Checks the value to see if it contains numerical characters or a "-"
 22      * 
 23      * @param value the string of characters to check
 24      * @return is a number (true) or not (false)
 25      */
 26     public static boolean isInteger(String value) {
 27         // Convert string into array of char values (built-in string method)
 28         char[] characters = value.toCharArray();
 29         // Loop (traverse) through this array from start to end
 30         for (int i = 0; i < characters.length; i++) {    
 31             // Use built-in method to see if this character is a digit (0-9)
 32             if (Character.isDigit(characters[i]) == false) {
 33                 // Character was not a digit, but is it the negative sign 
 34                 if (characters[i] != NEGATIVE) {
 35                     return false;       // Not a valid character for a integer
 36                 }
 37             }
 38         }
 39         return true;                    // All characters valid for a integer
 40     }

 44      * or a "."
 45      * 
 46      * @param value the string of characters to check
 47      * @return is a number (true) or not (false)
 48      */
 49     public static boolean isDouble(String value) {
 50         // Convert string into array of char values (built-in string method)
 51         char[] characters = value.toCharArray();
 52         // Loop (traverse) through this array from start to end
 53         for (int i = 0; i < characters.length; i++) {         
 54             // Use built-in method to see if this character is a digit (0-9)
 55             if (Character.isDigit(characters[i]) == false) {
 56                 // Character was not a digit, but is it the negative sign or
 57                 // the decimal symbol
 58                 if (characters[i] != DECIMAL && 
 59                     characters[i] != NEGATIVE) {
 60                     return false;       // Not a valid character for a double
 61                 }
 62             }
 63         }
 64         return true;                    // All characters valid for a double
 65     }

134     public static int[] random(int low, int high, int size) {
135         int[] numbers = new int[size];      // Create empty array of passed size
